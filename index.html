async function fetchAutoData() { const statusText = document.getElementById('statusText'); const refreshIcon = document.getElementById('refreshIcon'); refreshIcon.classList.add('fa-spin'); statusText.className = 'text-xs font-medium text-amber-400'; statusText.innerHTML = ` 正在擷取排列五最新數據...`; const timestamp = new Date().getTime(); const targetApi = `https://webapi.sporttery.cn/gateway/lottery/getHistoryPageListV1.qry?gameNo=350133&provinceId=0&pageSize=10&isVerify=1&pageNo=1&_=${timestamp}`; // 嘗試多個不同的免費跨域通道 const proxies = [ `https://api.codetabs.com/v1/proxy?quest=${encodeURIComponent(targetApi)}`, `https://api.allorigins.win/raw?url=${encodeURIComponent(targetApi)}` ]; let success = false; for (let proxyUrl of proxies) { try { // 設定 8 秒超時 const controller = new AbortController(); const timeoutId = setTimeout(() => controller.abort(), 8000); const response = await fetch(proxyUrl, { signal: controller.signal }); clearTimeout(timeoutId); if (!response.ok) continue; const realData = await response.json(); if (realData && realData.value && realData.value.list && realData.value.list.length > 0) { activeDatabase = realData.value.list.map(item => { const balls = item.lotteryDrawResult.trim().split(/\s+/).map(n => parseInt(n, 10)); return { period: item.lotteryDrawNum, balls: balls, date: item.lotteryDrawTime }; }); const now = new Date().toLocaleTimeString(); statusText.className = 'text-xs font-medium text-emerald-400'; statusText.innerHTML = ` 已成功同步最新 ${activeDatabase.length} 期真實數據！(${now})`; renderTable(activeDatabase); processAnalytics(activeDatabase); success = true; break; // 成功即跳出 } } catch (e) { console.warn('代理通道連線失敗，嘗試下一個...', e); } } if (!success) { const now = new Date().toLocaleTimeString(); statusText.className = 'text-xs font-medium text-rose-400'; statusText.innerHTML = ` 官方 API 連線超時，請點擊「重新同步」再試一次。(${now})`; } refreshIcon.classList.remove('fa-spin'); }
